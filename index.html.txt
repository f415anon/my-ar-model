<script>
const model = document.getElementById('myModel');

// Double-click for desktop
model.addEventListener('dblclick', () => {
    window.location.href = "https://sites.google.com/view/ammaribnasharif/";
});

// Double-tap for mobile
let lastTap = 0;
model.addEventListener('touchend', function(e) {
    let currentTime = new Date().getTime();
    let tapLength = currentTime - lastTap;
    if (tapLength < 500 && tapLength > 0) {
        window.location.href = "https://sites.google.com/view/ammaribnasharif/";
        e.preventDefault();
    }
    lastTap = currentTime;
});
</script>
